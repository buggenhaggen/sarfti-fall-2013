//Следующая программа MGR_EMP.CPP иллюстрирует использование наследования в C++ , создавая класс manager из базового класса employee. Программа определяет базовый класс employee, а затем определяет производный класс manager.Функция show_manager вызывает функцию show_employee, которая является элементом класса employee. Поскольку класс manager является производным класса employee, класс manager может обращаться к общим элементам класса employee, как если бы все эти элементы были определены внутри класса manager,


#include <iostream.h>

#include <string.h>

class employee

{ 
public: 
   employee(char *, char *, float); 
   void show_employee(void); 
private: 
   char name [ 64 ]; 
   char position[64]; 
   float salary; 
};

employee::employee(char *name, char *position,float salary)

{ 
   strcpy(employee::name, name); 
   strcpy(employee::position, position); 
   employee::salary = salary; 
}

void employee::show_employee(void)

{ 
   cout << "Имя: " << name << endl; 
   cout << "Должность: " << position << endl; 
   cout << "Оклад: $" << salary << endl; 
}

class manager : public employee

{ 
public: 
   manager(char *, char *, char *, float, float, int); 
   void show_manager(void); 
private: 
   float annual_bonus; 
   char company_car[64]; 
   int stock_options; 
};

manager::manager(char *name, char *position, char *company_car, float salary, float bonus, int stock_options) : employee(name, position, salary)

{ 
   strcpy(manager::company_car, company_car) ; 
   manager::annual_bonus = bonus ; 
   manager::stock_options = stock_options; 
}

void manager::show_manager(void)

{ 
   show_employee(); 
   cout << "Машина фирмы: " << company_car << endl; 
   cout << "Ежегодная премия: $" << annual_bonus << endl; 
   cout << "Фондовый опцион: " << stock_options << endl; 
}

void main(void)

{ 
   employee worker("Джон Дой", "Программист", 35000); 
   manager boss("Джейн Дой", "Вице-президент ", "Lexus", 50000.0, 5000, 1000); 
   worker.show_employee() ; 
   boss.show_manager(); 
}
